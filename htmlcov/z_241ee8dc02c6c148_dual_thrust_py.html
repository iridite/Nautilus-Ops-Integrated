37class DualThrustStrategyRefactored(BaseStrategy): 

38 """ 

39 Dual Thrust 突破策略 - 重构版本 

40 

41 使用模块化组件实现，代码更简洁清晰。 

42 

43 计算逻辑： 

44 - Range = Max(HH - LC, HC - LL) 

45 - 上轨 = Open + k1 * Range 

46 - 下轨 = Open - k2 * Range 

47 """

74 """处理 Bar 数据""" 

75 current_price = float(bar.close)

114 """检查入场信号"""

123 def _open_long(self, bar: Bar, current_price: float, upper_band: float | None): 

124 """开多仓""" 

125 qty = self.calculate_order_qty(bar.close) 

126 if qty and qty > 0: 

127 order = self.order_factory.market(

131 ) 

132 self.submit_order(order) 

133 self.position_opened = True 

134 self.log.info(f"突破上轨做多: price={current_price:.2f}, upper={upper_band:.2f}")

137 """开空仓"""

139 if qty and qty > 0:

144 ) 

145 self.submit_order(order) 

146 self.position_opened = True 

147 self.log.info(f"突破下轨做空: price={current_price:.2f}, lower={lower_band:.2f}")

155 """检查出场信号""" 

156 # 检查多头止损
