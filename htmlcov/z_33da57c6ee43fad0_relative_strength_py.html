7from collections import OrderedDict

10class RelativeStrengthCalculator: 

11 """ 

12 相对强度计算器 

13 

14 计算标的相对于基准（通常是 BTC）的相对强度： 

15 RS = (Symbol% - Benchmark%) 

16 其中 Symbol% 和 Benchmark% 是百分比收益率 

17 

18 支持多周期加权组合： 

19 Combined RS = short_weight * RS(short) + long_weight * RS(long) 

20 """ 

21 

22 def __init__( 

23 self, 

24 short_lookback_days: int = 5, 

25 long_lookback_days: int = 20, 

26 short_weight: float = 0.4, 

27 long_weight: float = 0.6, 

28 max_history_size: int = 500, 

29 ): 

30 """ 

31 初始化相对强度计算器 

32 

33 Args: 

34 short_lookback_days: 短期回溯天数 

35 long_lookback_days: 长期回溯天数 

36 short_weight: 短期权重 

37 long_weight: 长期权重 

38 max_history_size: 历史数据最大保存数量 

39 """ 

40 self.short_lookback_days = short_lookback_days 

41 self.long_lookback_days = long_lookback_days 

42 self.short_weight = short_weight 

43 self.long_weight = long_weight 

44 self.max_history_size = max_history_size 

45 

46 # 价格历史（使用 OrderedDict 保持插入顺序） 

47 self.symbol_price_history = OrderedDict() # {timestamp: price} 

48 self.benchmark_price_history = OrderedDict() # {timestamp: price} 

49 

50 def update_symbol_price(self, timestamp: int, price: float) -> None: 

51 """ 

52 更新标的价格历史 

53 

54 Args: 

55 timestamp: 时间戳（纳秒） 

56 price: 价格 

57 """ 

58 self.symbol_price_history[timestamp] = price 

59 

60 # 清理过期数据 

61 if len(self.symbol_price_history) > self.max_history_size: 

62 self.symbol_price_history.popitem(last=False) 

63 

64 def update_benchmark_price(self, timestamp: int, price: float) -> None: 

65 """ 

66 更新基准价格历史 

67 

68 Args: 

69 timestamp: 时间戳（纳秒） 

70 price: 价格 

71 """ 

72 self.benchmark_price_history[timestamp] = price 

73 

74 # 清理过期数据 

75 if len(self.benchmark_price_history) > self.max_history_size: 

76 self.benchmark_price_history.popitem(last=False) 

77 

78 def _get_common_timestamps(self) -> list[int] | None: 

79 """ 

80 获取标的和基准的共同时间戳 

81 

82 Returns: 

83 排序后的共同时间戳列表，如果数据不足则返回 None 

84 """ 

85 max_lookback = max(self.short_lookback_days, self.long_lookback_days) 

86 

87 if len(self.symbol_price_history) < max_lookback + 1: 

88 return None 

89 if len(self.benchmark_price_history) < max_lookback + 1: 

90 return None 

91 

92 common_timestamps = sorted( 

93 set(self.symbol_price_history.keys()) & set(self.benchmark_price_history.keys()) 

94 ) 

95 

96 if len(common_timestamps) < max_lookback + 1: 

97 return None 

98 

99 return common_timestamps 

100 

101 def _calculate_period_rs( 

102 self, common_timestamps: list[int], lookback_days: int 

103 ) -> float | None: 

104 """ 

105 计算指定周期的 RS 值 

106 

107 Args: 

108 common_timestamps: 共同时间戳列表 

109 lookback_days: 回溯天数 

110 

111 Returns: 

112 RS 值，如果数据不足或无效则返回 None 

113 """ 

114 # 边界检查：确保有足够的时间戳（需要 lookback_days + 1 个点） 

115 if len(common_timestamps) < lookback_days + 1: 

116 return None 

117 

118 period_ts = common_timestamps[-(lookback_days + 1) :] 

119 

120 # 边界检查：确保切片后仍有足够的数据点 

121 if len(period_ts) < lookback_days + 1: 

122 return None 

123 

124 symbol_prices = [self.symbol_price_history[ts] for ts in period_ts] 

125 benchmark_prices = [self.benchmark_price_history[ts] for ts in period_ts] 

126 

127 # 边界检查：确保价格列表不为空 

128 if not symbol_prices or not benchmark_prices: 

129 return None 

130 

131 # 检查价格有效性 

132 if symbol_prices[0] <= 0 or benchmark_prices[0] <= 0: 

133 return None 

134 

135 # 计算收益率 

136 symbol_perf = (symbol_prices[-1] / symbol_prices[0]) - 1 

137 benchmark_perf = (benchmark_prices[-1] / benchmark_prices[0]) - 1 

138 

139 # RS = Symbol% - Benchmark% 

140 return symbol_perf - benchmark_perf 

141 

142 def calculate_rs(self) -> float | None: 

143 """ 

144 计算加权组合 RS 分数 

145 

146 Returns: 

147 Combined RS = short_weight * RS(short) + long_weight * RS(long) 

148 如果数据不足则返回 None 

149 """ 

150 common_timestamps = self._get_common_timestamps() 

151 if common_timestamps is None: 

152 return None 

153 

154 # 计算短期 RS 

155 short_rs = self._calculate_period_rs(common_timestamps, self.short_lookback_days) 

156 if short_rs is None: 

157 return None 

158 

159 # 计算长期 RS 

160 long_rs = self._calculate_period_rs(common_timestamps, self.long_lookback_days) 

161 if long_rs is None: 

162 return None 

163 

164 # 加权组合 

165 combined_rs = self.short_weight * short_rs + self.long_weight * long_rs 

166 return combined_rs 

167 

168 def is_strong(self, threshold: float = 0.0) -> bool: 

169 """ 

170 判断标的是否相对强势 

171 

172 Args: 

173 threshold: RS 阈值（默认 0，表示跑赢基准） 

174 

175 Returns: 

176 True 表示标的相对强势 

177 """ 

178 rs = self.calculate_rs() 

179 return rs is not None and rs > threshold
