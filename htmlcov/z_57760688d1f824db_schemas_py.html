59 """ 

60

69 @classmethod 

70 def get_id_for( 

71 cls, 

72 venue_name: str, 

73 base_currency: str, 

74 quote_currency: str, 

75 inst_type: InstrumentType, 

76 ) -> str: 

77 """ 

78 根据交易所和货币对生成标准化的 instrument_id 

79 

80 Args: 

81 venue_name: 交易所名称（OKX, BINANCE） 

82 base_currency: 基础货币（USDT） 

83 quote_currency: 报价货币（BTC, ETH）

86 Returns:

88 

89 Raises: 

90 ValueError: 不支持的交易所

92 if venue_name == "OKX": 

93 return f"{quote_currency}-{base_currency}-{inst_type.value}.{venue_name}"

96 return f"{quote_currency}{base_currency}.{venue_name}" 

97 elif inst_type in (InstrumentType.SWAP, InstrumentType.PERP): 

98 return f"{quote_currency}{base_currency}-PERP.{venue_name}" 

99 elif inst_type in (InstrumentType.FUTURES, InstrumentType.OPTION): 

100 return f"{quote_currency}{base_currency}-{inst_type.value}.{venue_name}" 

101 else: 

102 raise ValueError(f"Unsupported venue: {venue_name}")

226 def _convert_params_to_dict(self) -> dict: 

227 """将params转换为字典""" 

228 if isinstance(self.params, dict): 

229 return self.params.copy() 

230 

231 if hasattr(self.params, "__dict__"): 

232 return {k: v for k, v in self.params.__dict__.items() if not k.startswith("_")} 

233 

234 try: 

235 from msgspec import structs 

236 

237 return structs.asdict(self.params) 

238 except (ImportError, TypeError): 

239 return {}

260 for label, bt in feed_bar_types.items(): 

261 if label not in ["main", "trend"]: 

262 p[f"{label}_bar_type"] = str(bt) 

263 

264 def resolve_params( 

265 self, 

266 instrument_id: Any, 

267 leverage: int, 

268 feed_bar_types: Dict[str, Any], 

269 ) -> Dict[str, Any]: 

270 """ 

271 解析并合并策略参数 

272

478 

479 if start_dt >= end_dt: 

480 raise ValueError("start_date must be before end_date")

828class ConfigPaths: 

829 """配置文件路径管理""" 

830 

831 def __init__(self, config_root: Optional[Path] = None): 

832 """ 

833 初始化配置路径管理器 

834 

835 Args: 

836 config_root: 配置文件根目录，默认为项目根目录下的 config 文件夹 

837 """ 

838 self.config_root = config_root or (project_root / "config") 

839 self.yaml_dir = self.config_root 

840 self.environments_dir = self.yaml_dir / "environments" 

841 self.strategies_dir = self.yaml_dir / "strategies" 

842 self.active_file = self.yaml_dir / "active.yaml" 

843 

844 def get_environment_file(self, env_name: str) -> Path: 

845 """获取环境配置文件路径""" 

846 return self.environments_dir / f"{env_name}.yaml" 

847 

848 def get_strategy_file(self, strategy_name: str) -> Path: 

849 """获取策略配置文件路径""" 

850 return self.strategies_dir / f"{strategy_name}.yaml" 

851 

852 def ensure_directories(self): 

853 """确保配置目录存在""" 

854 self.yaml_dir.mkdir(parents=True, exist_ok=True) 

855 self.environments_dir.mkdir(parents=True, exist_ok=True) 

856 self.strategies_dir.mkdir(parents=True, exist_ok=True) 

857 

858 def list_environments(self) -> List[str]: 

859 """列出所有可用的环境配置""" 

860 if not self.environments_dir.exists(): 

861 return [] 

862 

863 env_files = self.environments_dir.glob("*.yaml") 

864 return [f.stem for f in env_files if f.is_file()]

866 def list_strategies(self) -> List[str]: 

867 """列出所有可用的策略配置""" 

868 if not self.strategies_dir.exists(): 

869 return [] 

870 

871 strategy_files = self.strategies_dir.glob("*.yaml") 

872 return [f.stem for f in strategy_files if f.is_file()]
