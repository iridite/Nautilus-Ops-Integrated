213def _calculate_coverage_percentage( 

214 data_intervals: List[Tuple[int, int]], required_start: int, required_end: int 

215) -> float: 

216 """ 

217 计算数据覆盖率 

218 

219 Args: 

220 data_intervals: Parquet 数据的时间间隔列表 [(start_ns, end_ns), ...] 

221 required_start: 回测需要的开始时间（纳秒） 

222 required_end: 回测需要的结束时间（纳秒） 

223 

224 Returns: 

225 覆盖率百分比 (0.0 - 100.0) 

226 

227 算法： 

228 1. 计算所需的总时间范围 

229 2. 对于每个数据间隔，计算与所需范围的交集 

230 3. 累加所有交集的长度 

231 4. 覆盖率 = (交集总长度 / 所需总长度) * 100 

232 """ 

233 if not data_intervals: 

234 return 0.0 

235 

236 required_duration = required_end - required_start 

237 if required_duration <= 0: 

238 return 0.0 

239 

240 covered_duration = 0 

241 for interval_start, interval_end in data_intervals: 

242 # 计算交集 

243 overlap_start = max(interval_start, required_start) 

244 overlap_end = min(interval_end, required_end) 

245 

246 if overlap_start < overlap_end: 

247 covered_duration += overlap_end - overlap_start 

248 

249 coverage_pct = (covered_duration / required_duration) * 100.0 

250 return min(coverage_pct, 100.0) # 限制在 100% 以内

253def _check_parquet_coverage( 

254 catalog: ParquetDataCatalog, 

255 bar_type: BarType, 

256 cfg: BacktestConfig, 

257 stats: Optional[Dict[str, int]] = None, 

258 silent: bool = False, 

259) -> Tuple[bool, float, Optional[str]]: 

260 """ 

261 检查 Parquet 数据覆盖率 

262 

263 计算 Parquet 数据对回测时间范围的覆盖率，确保数据完整性。 

264 

265 Args: 

266 catalog: Parquet 数据目录 

267 bar_type: Bar 类型标识 

268 cfg: 回测配置（包含 start_date 和 end_date） 

269 stats: 统计信息字典（可选） 

270 silent: 是否静默模式（不输出日志，仅返回警告信息） 

271 

272 Returns: 

273 Tuple[bool, float, Optional[str]]: (是否满足要求, 覆盖率百分比, 警告信息) 

274 - 第一个值：覆盖率是否 >= 95% 

275 - 第二个值：实际覆盖率百分比 

276 - 第三个值：警告信息（仅当 coverage < 95% 且 silent=True 时返回） 

277 

278 Examples: 

279 >>> is_ok, pct, warning = _check_parquet_coverage(catalog, bar_type, cfg, silent=True) 

280 >>> if warning: 

281 ... print(warning) 

282 """ 

283 try: 

284 # 1. 获取数据间隔 

285 existing_intervals = catalog.get_intervals(data_cls=Bar, identifier=str(bar_type)) 

286 if not existing_intervals: 

287 return False, 0.0, None 

288 

289 # 2. 解析回测时间范围 

290 required_start, required_end = _parse_backtest_time_range(cfg) 

291 

292 # 3. 计算覆盖率 

293 coverage_pct = _calculate_coverage_percentage( 

294 existing_intervals, required_start, required_end 

295 ) 

296 

297 # 4. 判断是否满足要求（阈值 95%） 

298 COVERAGE_THRESHOLD = 95.0 

299 WARNING_THRESHOLD = 30.0 

300 is_sufficient = coverage_pct >= COVERAGE_THRESHOLD 

301 

302 # 5. 记录日志或返回警告信息 

303 warning_msg = None 

304 if not silent: 

305 # 立即输出日志（传统模式） 

306 if coverage_pct < WARNING_THRESHOLD: 

307 logger.warning( 

308 f"\nParquet coverage critically low for {bar_type}: " 

309 f"{coverage_pct:.1f}% < {WARNING_THRESHOLD}%" 

310 ) 

311 elif coverage_pct < COVERAGE_THRESHOLD: 

312 logger.info(f"Parquet coverage partial for {bar_type}: {coverage_pct:.1f}%") 

313 else: 

314 logger.debug(f"Parquet coverage OK for {bar_type}: {coverage_pct:.1f}%") 

315 else: 

316 # 静默模式：返回警告信息供收集 

317 if coverage_pct < COVERAGE_THRESHOLD: 

318 level = "critically low" if coverage_pct < WARNING_THRESHOLD else "partial" 

319 warning_msg = f"{bar_type}: {coverage_pct:.1f}% ({level})" 

320 

321 return is_sufficient, coverage_pct, warning_msg 

322 

323 except (KeyError, AttributeError) as e: 

324 if not silent: 

325 logger.warning(f"Failed to check coverage for {bar_type}: {e}") 

326 return False, 0.0, None

1138def _verify_data_consistency( 

1139 csv_path: Path, catalog: ParquetDataCatalog, bar_type: BarType 

1140) -> bool: 

1141 """ 

1142 验证 CSV 和 Parquet 数据的一致性（使用哈希校验） 

1143 

1144 使用 MD5 哈希值比较 CSV 文件内容，确保 Parquet 数据是从当前 CSV 导入的。 

1145 

1146 工作流程： 

1147 1. 检查 CSV 文件和 Parquet 数据是否存在 

1148 2. 计算当前 CSV 文件的哈希值 

1149 3. 从缓存获取上次导入时的哈希值 

1150 4. 比较哈希值判断是否一致 

1151 

1152 Args: 

1153 csv_path: CSV 文件路径 

1154 catalog: Parquet 数据目录 

1155 bar_type: 数据类型标识 

1156 

1157 Returns: 

1158 bool: 数据是否一致 

1159 - True: CSV 内容未变化，Parquet 数据可用 

1160 - False: CSV 已更新或首次导入，需要重新导入 

1161 """ 

1162 try: 

1163 # 1. 检查 CSV 文件基本信息 

1164 if not _check_csv_file_validity(csv_path): 

1165 return False 

1166 

1167 # 2. 检查 Parquet 数据是否存在 

1168 existing_intervals = _get_parquet_intervals(catalog, bar_type) 

1169 if not existing_intervals: 

1170 return False 

1171 

1172 # 3. 计算当前 CSV 文件哈希 

1173 current_hash = _calculate_csv_hash(csv_path) 

1174 

1175 # 4. 从缓存获取上次导入时的哈希 

1176 cached_hash = _get_cached_hash(csv_path, catalog) 

1177 

1178 # 5. 比较哈希值 

1179 if cached_hash is None: 

1180 # 首次导入，没有缓存 

1181 logger.debug(f"No cached hash for {csv_path.name}, will import") 

1182 return False 

1183 

1184 is_consistent = current_hash == cached_hash 

1185 

1186 if not is_consistent: 

1187 logger.info( 

1188 f"CSV data changed for {csv_path.name}: " 

1189 f"hash {cached_hash[:8]}... → {current_hash[:8]}..." 

1190 ) 

1191 else: 

1192 logger.debug(f"CSV data unchanged for {csv_path.name}, using cached Parquet") 

1193 

1194 return is_consistent 

1195 

1196 except (IOError, OSError) as e: 

1197 logger.warning(f"Hash verification failed for {csv_path.name}: {e}") 

1198 # 验证失败时，保守地返回 False，触发重新导入 

1199 return False

1601def _build_filter_stats(filter_stats: Optional[Dict[str, Dict[str, int]]]) -> Dict[str, Any]: 

1602 """构建过滤器统计""" 

1603 if not filter_stats: 

1604 return {}

1606 total_stats: Dict[str, Any] = {} 

1607 for inst_id, stats in filter_stats.items(): 

1608 for key, value in stats.items(): 

1609 total_stats[key] = total_stats.get(key, 0) + value 

1610 

1611 signal_checks = total_stats.get("signal_checks", 0) 

1612 all_passed = total_stats.get("all_passed", 0) 

1613 

1614 filter_rates = {} 

1615 if signal_checks > 0: 

1616 filter_rates = { 

1617 "pass_rate": (all_passed / signal_checks * 100), 

1618 "filter_rate": ((signal_checks - all_passed) / signal_checks * 100),

1620 for key in [ 

1621 "fail_trend", 

1622 "fail_squeeze", 

1623 "fail_squeeze_maturity", 

1624 "fail_rs", 

1625 "fail_extension", 

1626 "fail_btc_bear", 

1627 "fail_breakout", 

1628 "fail_volume", 

1629 ]: 

1630 if key in total_stats: 

1631 filter_rates[f"{key}_rate"] = total_stats[key] / signal_checks * 100 

1632 

1633 return { 

1634 "by_instrument": filter_stats, 

1635 "total": total_stats, 

1636 "rates": filter_rates,
